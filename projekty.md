double tyzdenna_mzda(double h_mzda, double hod)
{
	double p,p2;
	p = hod - 40;
	p2 = hod - 60;

	if (hod < 40)
		return(h_mzda * hod);
	
	else if (hod >=40 && hod <= 60)
		return((h_mzda * 40) + h_mzda * p * 1.5);
	
	else 
		return((h_mzda * 40) + (h_mzda * 20 * 1.5) + (h_mzda * p2 * 2));
}

	int n,i;
	double m,h,c;

	scanf("%d",&n);

		c += tyzdenna_mzda(m,h);
	}

return 0;
